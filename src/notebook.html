from math import floor, ceil

def round_rating_to_nearest_even(rating):
    """
    7.36 -> 7
    7.64 -> 7.5
    7.5 -> 7.5
    6.5 -> 6.5
    6.4 -> 6
    6.6 -> 7
    """
    if rating % 1 == 0:
        return rating
    elif 0 < rating % 1 <= 0.25:
        return floor(rating)
    elif 0.25 < rating % 1 <= 0.75:
        return floor(rating) + 0.5
    else:
        return ceil(rating)
